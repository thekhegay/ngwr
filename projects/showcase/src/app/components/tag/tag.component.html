API

Property Description Type Default
[color] set color of tag 'primary' | 'secondary' | 'success' | 'warning' | 'danger' | 'light' | 'medium' | 'dark' 'primary'
[outlined] set outlined style for tag boolean false
[rounded] set rounded style for tag boolean false
[transparent] set transparent style for tag boolean false
[hoverable] set hoverable style for tag boolean false
[icon] add icon to tag wrIconName null
[iconPosition] set position of icon 'start' | 'end' 'start'
[loading] set loading state for tag boolean false
